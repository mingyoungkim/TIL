arr = [55, 7, 78, 12, 42]

def BubbleSort(a):
    for i in range(len(a)-1, 0, -1):

        # 한 단계마다 정렬은 마지막 요소가 되므로
        # i 를 4, 3, 2, 1 으로 만들어서 범위를 잡아주자
        # 마지막에 남는 a[0] 은 정렬할 필요 없으니까 4번하자
        for j in range(0, i):
            if a[j] > a[j+1]:
                a[j], a[j+1] = a[j+1], a[j]
    return a

> 카운팅 정렬

```python
arr = [0, 4, 1, 3, 1, 2, 4, 1]

def CountingSort(A):
    # A 는 원본 데이터 (정렬될 배열)
    
    # K 는 cnt 의 가장 큰 수
    k = 0
    for i in range(len(A)):
        if A[i] > k:
            k = A[i]

    cnt = [0] * (k+1)

    for i in range(len(A)):
        cnt[A[i]] += 1

    # cnt 의 처음 0번째는 누적 안해도 되니까 시작이 1
    for i in range(1, len(cnt)):
        cnt[i] += cnt[i-1]

    # B 는 정렬하고 넣을 입력 배열
    B = [0] * len(A)

    for i in range(len(A)-1, -1, -1):
        B[cnt[A[i]]-1] = A[i]
        cnt[A[i]] -= 1
    return B

N_arr = CountingSort(arr)
print(N_arr)
```
