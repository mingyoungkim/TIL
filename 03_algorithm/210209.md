arr = [0, 4, 1, 3, 1, 2, 4, 1]

def CountingSort(A):
    # A ëŠ” ì›ë³¸ ë°ì´í„° (ì •ë ¬ë  ë°°ì—´)
    
    # k ëŠ” cnt ì˜ ê°€ì¥ í° ìˆ˜

            k = A[i]

    for i in range(len(A)):
        cnt[A[i]] += 1

    # cnt ì˜ ì²˜ìŒ 0ë²ˆì§¸ëŠ” ëˆ„ì  ì•ˆí•´ë„ ë˜ë‹ˆê¹Œ ì‹œì‘ì´ 1

        cnt[A[i]] -= 1
    return B
