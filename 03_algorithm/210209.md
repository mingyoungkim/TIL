> 카운팅 정렬

```python
arr = [0, 4, 1, 3, 1, 2, 4, 1]

def CountingSort(A):
    # A 는 원본 데이터 (정렬될 배열)
    
    # K 는 cnt 의 가장 큰 수
    k = 0
    for i in range(len(A)):
        if A[i] > k:
            k = A[i]

    cnt = [0] * (k+1)

    for i in range(len(A)):
        cnt[A[i]] += 1

    # cnt 의 처음 0번째는 누적 안해도 되니까 시작이 1
    for i in range(1, len(cnt)):
        cnt[i] += cnt[i-1]

    # B 는 정렬하고 넣을 입력 배열
    B = [0] * len(A)

    for i in range(len(A)-1, -1, -1):
        B[cnt[A[i]]-1] = A[i]
        cnt[A[i]] -= 1
    return B

N_arr = CountingSort(arr)
print(N_arr)
```

### 02. Baby-gin game

```
0~9 까지의 숫자 카드에서 임의의 카드 6장을 뽑자.

그 때, 3장의 카드가 연속적인 번호를 가지면 run

3장의 카드가 동일한 숫자를 가지면  triplet

그리고 6장의 카드가 run과 triplet으로 구성된 경우, baby-gin
```



> 완전검색으로 Baby-gin 구현

```python
num = [4, 5, 8, 0, 6, 7]

# run과 triplet 변수 초기화
run = False
tri = False

for i in range(len(num)):
    for j in range(len(num)):
        
        # num에 같은 인덱스 숫자가 겹치지 않도록
        if i != j:
            for k in range(len(num)):
                # i, j, k역시 다 다른 인덱스의 숫자여야한다.
                if k != i and k != j:
                    # 다 다른 인덱스의 숫자를 출력해보자
                    print(num[i], num[j], num[k])
                    
                    # run이면 True로 바꾸자
                    if num[i]+1 == num[j] and num[i]+2 == num[k]:
                        run = True
                        print('run')
                        
                    # triplet이 나오면 True로 바꾸자
                    if num[i] == num[j] == num[k]:
                        tri = True
                        print('triplet')
                        
# 3개씩 나눠서 run과 triplet을 검사한 결과
# 모든 경우의 수에서 run과 triplet이 있으면
# 즉, True가 되었다면
# baby-gin이다
if run and tri:
    print('baby-gin')
```



> 카운팅 정렬로 baby-gin 구현하기

```python
# 입력들어 온 숫자
num = 555789
cnt = [0] * 10

# int 함수 주의 !!!!
# 만약 들어온 문자가 050072 인데
# int('050072')입력하면
# 내장함수 성격상 맨 앞의 0 이 사라짐

for i in range(6):
    # 숫자가 담긴 리스트가 아닌
    # 연결된 숫자로 입력되어 있기 때문에
    # 10으로 나눠 나머지를 빈 cnt에 더해준다
    cnt[num % 10] += 1
    # num을 10 으로 나눈 몫으로 차곡차곡 바꾸자
    num //= 10

n = 0
run = tri = 0
while n < 10:
    
    # triplet
    if cnt[n] >= 3:
        cnt[n] -= 1
        tri += 1
        
    # run
    # 이때, n이 8이상이면
    # cnt[n+1], cnt[n+2]가 inx범위를 벗어나기때문에
    # n 조건을 다시 설정
    if n < 8:
        if cnt[n] >= 1 and cnt[n+1] >= 1 and cnt[n+2] >= 1:
            cnt[n] -= 1
            cnt[n+1] -= 1
            cnt[n+2] -= 1
            run += 1
    n += 1

# run과 triplet 둘 다 1 이상이면
# baby-gin
if run >= 1 and tri >= 1:
    print('baby-gin')
else:
    print('none')
```
